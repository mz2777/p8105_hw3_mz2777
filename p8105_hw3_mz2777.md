``` r
# scatterplot for the number of items ordered in each aisle
instacart %>%
  group_by(aisle) %>%
  summarize(n_obs = n()) %>% 
  filter(n_obs > 10000) %>% 
  ggplot(aes(x = aisle, y = sort(n_obs))) + 
  geom_point() + 
  labs(
    title = "The Number of Items Ordered in each Aisle",
    x = "Aisle Name",
    y = "The number of items"
```
